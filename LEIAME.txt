# Projeto 4 - Automação de Pipeline Para Extração e Armazenamento de Dados com Selenium

# Abra o terminal ou prompt de comando, navegue até a pasta com os arquivos e execute o comando abaixo para criar um ambiente virtual:

conda create --name dsaprocp4 python=3.13

# Ative o ambiente:

conda activate dsaprocp4 (ou: source activate dsaprocp4)

# Instale o pip e as dependências:

conda install pip
pip install -r requirements.txt 

# Execute os comandos abaixo:

python projeto4_v1.py
python projeto4_v2.py

# Use os comandos abaixo para desativar o ambiente virtual e remover o ambiente (opcional):

conda deactivate
conda remove --name dsaprocp4 --all
